"modernize (disregard vi compatibility)
set nocompatible

"turn syntax highlighting on
syntax on

"enable plugins (for netrw)
filetype plugin on

"; = insertion mode (instead of i)
noremap ; i

"j = left  (instead of h)
"k = down  (instead of j)
"i = up    (instead of k)
"l = right (already default)
"h = none  (previously left)
noremap j h
noremap k j
noremap i k
noremap h <nop>

"enable line numbers
set number

"show status bar
set laststatus=2
set statusline=%f "tail of the filename
